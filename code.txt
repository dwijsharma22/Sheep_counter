int pin1=5;
int pin2=9;

int sensorState1 = 0;
int sensorState2 = 0;
int count=0;


void setup()
{
  pinMode(pin1, INPUT);
  pinMode(pin2, INPUT);
  
  Serial.begin(9600);
}

void loop()
 {  
  sensorState1 = digitalRead(pin1);
  sensorState2 = digitalRead(pin2);

  if(sensorState1 == LOW){
    count++; 
    delay(500);
  }

  if(sensorState2 == LOW){
    count--; 
    delay(500);
  }
  
  
  Serial.println("No. of sheeps: ");
  Serial.println(count);
  delay(200);
}
